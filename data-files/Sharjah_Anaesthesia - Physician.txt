    var ValidateECaptchaEnquiry = function () {

    String.prototype.hashCode = function () {
        var hash = 5381;
        if (this.length == 0) return hash;
        for (i = 0; i < this.length; i++) {
            char = this.charCodeAt(i);
            hash = ((hash << 5) + hash) + char;
            hash = hash & hash; // Convert to 32bit integer
        }
        return hash;
    }
    function validateEmail(email) {
        if (email && email != "") {
            var re = /\w+([-+.']\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*/;
            if (!re.test(email)) {
                $("#lblemailValMessage").text("Please enter a valid email address. Ex: abc@xyz.com");
            }
            else {
                $("#lblemailValMessage").text("");
            }
            return re.test(email);
        } else {
            $("#lblemailValMessage").text("This field is required.");
            return false;
        }
    }

     }

     function validatePhoneNumber(txtPhone) {
         if ($.trim(txtPhone) != '') {

                 $("#lblPhoneVal").text("Please enter a valid phone number.");

                  $("#lblPhoneVal").text("");
              }

          }
          else {
              $("#lblPhoneVal").text("");
              return true;
          }
      }
